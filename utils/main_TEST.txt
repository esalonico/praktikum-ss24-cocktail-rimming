#include <Arduino.h>

// global variables (macros)
#define LED 12
#define MOTOR 17

// runs only once at setup
void setup()
{
    Serial.begin(115200);
    Serial.println("Setup done!");

    // setup led
    pinMode(LED, OUTPUT);

    // setup motor
    pinMode(MOTOR, OUTPUT);
}

void loop()
{
    digitalWrite(LED, 1);
    // digitalWrite(MOTOR, 1);
    Serial.println("Led and motor ON");
    delay(2000);

    digitalWrite(LED, 0);
    // digitalWrite(MOTOR, 0);
    Serial.println("Led and motor OFF");

    delay(2000);
}
