#include <Arduino.h>

// global variables (macros)
#define LED 19
#define MOTOR 17

// runs only once at setup
void setup()
{
    Serial.begin(115200);
    Serial.println("Setup done!");

    // setup led
    pinMode(LED, OUTPUT);

    // setup motor
    pinMode(MOTOR, OUTPUT);
}

void loop()
{
    for (int i = 0; i < 256; i++)
    {
        analogWrite(MOTOR, i);
        delay(10);
    }

    digitalWrite(LED, 1);
    delay(2000);

    for (int i = 255; i >= 0; i--)
    {
        analogWrite(MOTOR, i);
        delay(10);
    }
    digitalWrite(LED, 0);

    // digitalWrite(LED, 1);

    // Serial.println("Led and motor OFF");

    // delay(1000);
}
